//En Javascript le scope est lié aux fonctions, pas aux blocs de code


function maSuperFonction(items){
    for (var i=0; i<items.length; i++){
        var hello = 'world ' + i;
        setImmediate(function(){

            //attention, la variable hello est en fait définie
            //au niveau de la fonction !

            console.log('hello ' + hello);
        });
    }
}

maSuperFonction([0,1,2,3,4]);
